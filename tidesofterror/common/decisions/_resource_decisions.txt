    develop_steel_deposits = { 

		icon = steel

		allowed = {

		}

		available = {
			has_tech = indtech_shaft_mining
			num_of_civilian_factories_available_for_projects > 2
			any_owned_state = {
				steel > 0
            }
		}

		visible = {
			any_owned_state = {
				steel > 0
            }
		}

		fire_only_once = no

		cost = 25
		days_remove = 120

		modifier = {
			civilian_factory_use = 3
		}

		ai_will_do = {
			base = 0.3
			modifier = {  # Too low PP
				factor = 0.5
					has_political_power < 80
			}
			modifier = { #We have plenty of PP, avanti!
				factor = 3
				has_political_power > 200
			}
			modifier = { #Running out of the stuff, need more!
				factor = 5
				has_resources_in_country = {
					resource = steel
					amount < 1
				}
			}
		}
		remove_effect = {
            custom_effect_tooltip = develop_steel_deposits_tt
            hidden_effect = {
                random_owned_state = {
                    limit = {
                        steel > 0
                    }
                    add_resource = {
                        type = steel
                        amount = 3
                    }
                }
            }
		}
	}
